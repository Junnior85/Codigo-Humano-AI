@app.route('/', methods=['GET', 'POST'])
def home():
    mensaje = None
    color = "#888"
    
    if request.method == 'POST':
        texto = request.form.get('contenido')
        modo = request.form.get('modo')
        
        if texto:
            fecha_hora = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            
            if modo == 'guardar':
                # Opción 1: Guardar en disco
                with open("bitacora_web.txt", "a", encoding="utf-8") as f:
                    f.write(f"[{fecha_hora}] {texto}\n")
                mensaje = "DATOS ESCRITOS EN DISCO CORRECTAMENTE."
                color = "#4caf50" # Verde
                
            elif modo == 'volatil':
                # Opción 2: El desahogo. No guardamos nada.
                # Técnicamente recibimos el dato y dejamos que la variable muera.
                mensaje = "BUFFER DE RAM LIMPIADO. DATOS ELIMINADOS."
                color = "#e53935" # Rojo suave

    # Leemos el historial para mostrarlo abajo (solo las ultimas lineas)
    historial = ""
    if os.path.exists("bitacora_web.txt"):
        with open("bitacora_web.txt", "r", encoding="utf-8") as f:
            lines = f.readlines()
            historial = "".join(lines[-5:]) # Muestra solo las últimas 5 entradas

    return render_template_string(HTML_TEMPLATE, mensaje=mensaje, color=color, historial=historial)

if __name__ == '__main__':
    # debug=True permite ver errores si algo falla
    app.run(debug=True, port=5000)
